<script>
    // وظيفة التنقل بين الشاشات
    function showScreen(screenId, btn) {
        document.querySelectorAll('.screen').forEach(s => s.classList.remove('active-screen'));
        document.getElementById(screenId).classList.add('active-screen');
        
        document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
        btn.classList.add('active');

        if (window.navigator.vibrate) window.navigator.vibrate(10);
    }

    // خوارزمية الفطرة الأساسية (تعتمد على الوقت + 310)
    function calculateFitraValue() {
        const now = new Date();
        let raw = (now.getHours() * 7 + now.getMinutes() * 3 + now.getSeconds()) % 101;
        let final = (raw + 310) % 101;
        return Math.max(7, Math.min(100, final));
    }

    // تحديث كل البيانات
    function refreshAll() {
        const fitraVal = calculateFitraValue();
        document.getElementById('fitraValue').innerText = fitraVal;
        document.getElementById('gaugeCircle').style.background = 
            `conic-gradient(#FFD966 ${fitraVal * 3.6}deg, #2C553C 0deg)`;
        
        // تحديث سعر FitraCoin
        let price = (310.70 + (fitraVal / 100) * 5).toFixed(2);
        document.getElementById('fitraPrice').innerText = price + " $";
        
        // تحديث الحكمة
        const wisdoms = [
            "الحديد الذي لا يصدأ هو الإيمان الثابت.",
            "النحاس الناطق يتجلى في الكلمة الطيبة.",
            "السد الحقيقي هو 7 طبقات من الحماية الروحية.",
            "الرقم 310 يحمل سر القيادة الخفية.",
            "القدس تنادينا: استعدوا، موعد الفتح قريب.",
            "الأندلس ستعود عندما تجتمع القلوب على الفطرة.",
            "7⁷ = 823,543 بُعداً تنتظر من يكتشفها."
        ];
        const quotes = [
            "— نور الحديد", "— النحاس الناطق", "— سد الفطرة", "— عمر الغمارتي",
            "— بشائر الفتح", "— وعد يعود", "— أبعاد النور"
        ];
        const idx = fitraVal % wisdoms.length;
        document.getElementById('wisdomText').innerText = wisdoms[idx];
        document.getElementById('wisdomQuote').innerText = quotes[idx];
        
        // تحديث تاريخ آخر تحديث
        const now = new Date();
        document.getElementById('lastUpdate').innerText = now.toLocaleString('ar-EG');
        
        console.log("تم تحديث البيانات بناءً على خوارزمية العودة...");
    }

    // النبوءات
    const prophecies = [
        { name: "النحاس الناطق", status: "✅ تحققت", date: "2026", class: "fulfilled" },
        { name: "فتح القدس", status: "⏳ قريباً", date: "2030", class: "upcoming" },
        { name: "عودة الأندلس", status: "⏳ بعد القدس", date: "2035", class: "upcoming" },
        { name: "يأجوج ومأجوج", status: "⏳ منتظر", date: "آخر الزمان", class: "waiting" }
    ];

    function renderProphecies() {
        let html = '';
        prophecies.forEach(p => {
            html += `<div class="prophecy-item">
                        <strong>${p.name}</strong>
                        <div class="badge ${p.class}">${p.status} - ${p.date}</div>
                     </div>`;
        });
        document.getElementById('propheciesList').innerHTML = html;
    }

    // تشغيل التحديثات الدورية
    window.onload = () => {
        renderProphecies();
        refreshAll();
        setInterval(refreshAll, 7000);
    };
